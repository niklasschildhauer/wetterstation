/**
 * Model of the User Identifikation. It contains the token 
 * and user id. These are needed for the identification of the user 
 * by the backend. 
 */
export interface UserIdentifikation {
    token: string,
    id: number,
}

export enum Themes {
    Light,
    Dark,
    Automatic,
    HighContrast
};

export interface PollenType {
    id: number,
    pollenName: string
};

export enum Language {
    'de-DE',
    'en-EN'
};
